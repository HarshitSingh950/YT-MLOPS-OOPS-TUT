EXPERIMENT 1
mkdir mlops-unit1
cd mlops-unit1
git init
git remote add origin https://github.com/YOUR-USERNAME/mlops-unit1.git
git add .
git commit -m "Initial commit: Dataset loading and preprocessing"
git branch -M main
git push -u origin main

Create experiment branch:
git checkout -b experiment-v1

After editing preprocess.py and/or adding correlation output:
git add src\preprocess.py
git commit -m "Experiment: Added correlation analysis"
git push origin experiment-v1

Merge experiment branch:
git checkout main
git merge experiment-v1
git push origin main
